Ewl

The Enlightenment Widget Library, a simple-to-use general purpose widget library based on the Evas canvas. This library is still in development but already has demo applications and solid documentation, so feel free to try the CVS sources.

Here's an example of Ewl usage:

#include <Ewl.h>

/* Define the main window handle to EWL */
Ewl_Widget *main_win;

int
main(int argc, char **argv)
{
        /* Define a button handle to EWL */
        Ewl_Widget *button;

        /* Initialize EWL */
        ewl_init(argc, argv);

        /* Setup and show the main window */
        main_win = ewl_window_new();
        ewl_window_set_title(main_win,
                             "The Cuddletech Test App: EWL");
        ewl_window_resize(main_win, 297, 68); 
        ewl_widget_show(main_win);

        /* Setup and show a button */
        button = ewl_button_new("Kool Button");
	ewl_object_set_custom_size(EWL_OBJECT(button), 100, 17);
        ewl_container_append_child(EWL_CONTAINER(main_win), button);
        ewl_widget_show(button);

        /* Go into the main EWL Event Loop */
        ewl_main();

        return 1;
}
