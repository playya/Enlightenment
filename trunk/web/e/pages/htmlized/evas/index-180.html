static void
mouse_down(void *_data, Evas _e, Evas_Object _o, int _b, int _x, int _y)
{ double x, y;
evas_get_geometry(_e, _o, &x, &y, NULL, NULL);
evas_move(_e, _o, x + 16.0, y + 16.0);
}
static void

{ double x, y;
evas_get_geometry(_e, _o, &x, &y, NULL, NULL);
evas_move(_e, _o, x - 16.0, y - 16.0);

}

double
get_time(void)
{ struct timeval timev;

}
